# Log in, log out

# Uses the pre-configured OpenShift cluster
oc login

# Allows you to log in to any OpenShift cluster
oc login <cluster address>

# Log out
oc logout


# Project Basics

# See current project
oc project

# Create a new project
oc new-project demo-project

# List all projects
oc projects

# Switch projects
oc project <project name>
